## Installation Script

```bash
#!/bin/bash
# install_v5_deps.sh

cd /home/ml

# Create virtual environment
python3 -m venv venv_v5
source venv_v5/bin/activate

# Install Temporal
pip install temporalio==1.4 temporal-sdk-core==1.4

# Install YouTube API
pip install google-api-python-client
pip install google-auth-httplib2
pip install google-auth-oauthlib

# Install ClawTeam
cd /home/ml/ClawTeam-OpenClaw
pip install -e .
cd /home/ml

# Install DSPy
pip install dspy-ai langgraph

# Install real-time
pip install openai-whisper librosa soundfile

# Install monitoring
pip install prometheus-client structlog python-json-logger

# Install remaining deps
pip install qdrant-client neo4j pydantic>=2.6.0,<3.0.0 numpy pandas scipy requests tqdm

echo "v5.0 dependencies installed successfully"
```

## Docker Installation

```dockerfile
FROM python:3.11-slim

WORKDIR /app

# Install system dependencies
RUN apt-get update && apt-get install -y \
    gcc \
    && rm -rf /var/lib/apt/lists/*

# Copy requirements and install
COPY requirements_v5.txt .
RUN pip install --no-cache-dir -r requirements_v5.txt

# Copy application code
COPY smb_knowledge_base_v5/ ./smb_knowledge_base_v5/
COPY ClawTeam-OpenClaw/ ./ClawTeam-OpenClaw/
RUN pip install -e ./ClawTeam-OpenClaw/

# Create entrypoint
CMD ["python", "-m", "smb_knowledge_base_v5.orchestration.worker"]
```
